```javascript
let b = 137.5;
let c = 10;
let n = 0;

function setup() {
    createCanvas(windowWidth, windowHeight);
    angleMode(DEGREES);
    background(0);
}

function draw() {
    let a = n * b;
    let r = c * sqrt(n);
    let x = r * cos(a) + width / 2;
    let y = r * sin(a) + height / 2;
    noStroke();
    fill((tan(n) * 128), (sin(n) * 128), (atan(n) * 128));
    ellipse(x, y, 20);
    b += 0.005;
    n++;
}
```
